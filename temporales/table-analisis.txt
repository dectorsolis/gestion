CREATE TABLE op_analisis_prospecto(
	id_analisis INTEGER AUTO_INCREMENT PRIMARY KEY,
	id_prospecto INTEGER NOT NULL,
	id_empleado INTEGER,
	fecha_creacion DATE NOT NULL,
	fecha_actualizacion DATE,
	observaciones TEXT,
	rating smallint,
	status boolean NOT NULL,
	FOREIGN KEY(id_prospecto) REFERENCES prospecto(id_prospecto) ON UPDATE CASCADE ON DELETE CASCADE,
	FOREIGN KEY(id_empleado) REFERENCES empleado(id_empleado) ON UPDATE CASCADE ON DELETE CASCADE
);
